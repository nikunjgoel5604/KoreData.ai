<script>
    // ═══════════════════════════════════════════
    // GLOBAL STATE
    // ═══════════════════════════════════════════
    let globalData = null;
    let currentChart = null;
    let corrChart = null;
    let dashboardItems = [];
    let workspaces = JSON.parse(localStorage.getItem('dataviz_workspaces')) || [];
    let currentWorkspace = null;

    // ═══════════════════════════════════════════
    // AUTH STATE
    // ═══════════════════════════════════════════
    let authToken   = null;
    let authUser    = null;

    // ═══════════════════════════════════════════
    // CODE PANEL — defined first so FAB onclick works
    // ═══════════════════════════════════════════
    function openCodePanel(runSim) {
        const panel    = document.getElementById('codePanel');
        const backdrop = document.getElementById('codePanelBackdrop');
        if (!panel || !backdrop) return;
        panel.classList.add('open');
        backdrop.classList.add('open');
        if (runSim) setTimeout(() => triggerFullSimulation(), 50);
    }

    function closeCodePanel() {
        const panel    = document.getElementById('codePanel');
        const backdrop = document.getElementById('codePanelBackdrop');
        if (!panel || !backdrop) return;
        panel.classList.remove('open');
        backdrop.classList.remove('open');
    }

    function setPhase(label, pct) {
        const el   = document.getElementById('simPhaseLabel');
        const fill = document.getElementById('simProgressFill');
        if (el)   el.textContent = label;
        if (fill) fill.style.width = pct + '%';
    }

    function addCodeLine(text, cls, container) {
        const el = document.createElement('div');
        el.className = 'code-line' + (cls ? ' ' + cls : '');
        el.textContent = text;
        container.appendChild(el);
        container.parentElement.scrollTop = container.parentElement.scrollHeight;
    }

    function triggerFullSimulation() {
        const box = document.getElementById('codePanelContent');
        if (!box) return;
        box.innerHTML = '';
        setPhase('// Initializing...', 0);
        if (globalData) {
            runRealDataSimulation(globalData, box);
        } else {
            runDemoSimulation(box);
        }
    }

    // ═══════════════════════════════════════════
    // MODAL FUNCTIONS
    // ═══════════════════════════════════════════
    function showCreateWorkspaceModal() {
        document.getElementById('workspaceModal').classList.add('active');

    function createWorkspace() {
        const name = document.getElementById('wsName').value || 'Untitled';
        const desc = document.getElementById('wsDesc').value || '';

        const workspace = {
            id: Date.now(),
            name: name,
            description: desc,
            createdAt: new Date().toLocaleString(),
            data: null
        };

        workspaces.push(workspace);
        localStorage.setItem('dataviz_workspaces', JSON.stringify(workspaces));

        document.getElementById('wsName').value = '';
        document.getElementById('wsDesc').value = '';
        closeModal('workspaceModal');
        renderWorkspaces();
        showToast('✅ Workspace created');
    }

    function renderWorkspaces() {
        const container = document.getElementById('workspacesContainer');
        if (!container) return;

        container.innerHTML = '';
        if (workspaces.length === 0) {
            container.innerHTML = '<p style="color: var(--text-dim); text-align: center; padding: 40px;">No workspaces yet. Create one!</p>';
            return;
        }

        workspaces.forEach(ws => {
            const card = document.createElement('div');
            card.className = 'card';
            card.innerHTML = `
                <div style="display: flex; justify-content: space-between; align-items: flex-start; margin-bottom: 12px;">
                    <div style="flex: 1;">
                        <h3 style="font-size: 16px; margin-bottom: 8px;">${ws.name}</h3>
                        <p style="color: var(--text-dim); font-size: 13px;">${ws.description}</p>
                    </div>
                </div>
                <div style="display: flex; gap: 8px;">
                    <button class="btn btn-primary btn-sm" onclick="openWorkspace(${ws.id})" style="flex: 1;">Open</button>
                    <button class="btn btn-outline btn-sm" onclick="deleteWorkspace(${ws.id})">🗑️</button>
                </div>
            `;
            container.appendChild(card);
        });
    }

    function openWorkspace(wsId) {
        currentWorkspace = workspaces.find(w => w.id === wsId);
        if (currentWorkspace) {
            document.getElementById('currentWorkspace').textContent = currentWorkspace.name;
            showToast(`📂 ${currentWorkspace.name}`);
            showSection('upload');
        }
    }

    function deleteWorkspace(wsId) {
        if (confirm('Delete this workspace?')) {
            workspaces = workspaces.filter(w => w.id !== wsId);
            localStorage.setItem('dataviz_workspaces', JSON.stringify(workspaces));
            renderWorkspaces();
            showToast('🗑️ Deleted');
        }
    }

    function showSection(name) {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

        const sec = document.getElementById('section-' + name);
        if (sec) sec.classList.add('active');

        const navItem = document.querySelector(`[data-section="${name}"]`);
        if (navItem) navItem.classList.add('active');

        // Re-render sections on navigation if data is already loaded
        // This ensures sections show data even if user navigated before upload completed
        if (globalData) {
            try {
                if (name === 'viewdata') {
                    if (_vdData !== globalData) initViewDataset(globalData);
                    else vdRenderTable();
                } else if (name === 'overview')     renderOverview(globalData);
                else if (name === 'quality')         renderQuality(globalData);
                else if (name === 'structure')       renderStructure(globalData);
                else if (name === 'charts')          renderCharts(globalData);
                else if (name === 'correlation')     renderCorrelation(globalData);
                else if (name === 'insights')        renderInsights(globalData);
                else if (name === 'pivot')           renderPivot(globalData);
                else if (name === 'values')          renderValueCounts(globalData);
                else if (name === 'predict')         populatePredictModel(globalData);
            } catch(e) { console.warn('Section re-render error:', e); }
        }
    }

    function runRealDataSimulation(d, box) {
        const ov       = d.overview;
        const mp       = d.missing_handling_process;
        const stats    = d.statistics || {};
        const outliers = d.data_quality?.outliers || {};
        const corr     = d.advanced_visualization?.correlation || {};
        const vc       = d.value_counts || {};
        const hists    = d.visualization?.histograms || {};

        const numCols  = ov.numeric_columns;
        const catCols  = ov.categorical_columns;
        const dtCols   = ov.datetime_columns;
        const allCols  = [...numCols, ...catCols, ...dtCols];

        const phases = [
            // Phase 1 – Import + Load
            { delay: 0, phase: '[ 1/7 ] Importing Libraries', pct: 5, lines: [
                ['import pandas as pd', 'keyword'],
                ['import numpy as np', 'keyword'],
                ['import matplotlib.pyplot as plt', 'keyword'],
                ['import seaborn as sns', 'keyword'],
                ['from eda_engine import perform_eda', 'keyword'],
                ['', ''],
                ['# ── Loading your dataset ─────────────', 'dim'],
                [`df = pd.read_csv('your_dataset.csv')`, 'value'],
                [`# → Shape: (${ov.rows.toLocaleString()}, ${ov.columns})`, 'value'],
                ['✔ Dataset loaded successfully', 'success'],
            ]},

            // Phase 2 – Shape & dtypes
            { delay: 1000, phase: '[ 2/7 ] Shape & Column Types', pct: 20, lines: [
                ['', ''],
                ['# ── df.shape & dtypes ───────────────', 'dim'],
                [`print(df.shape)   →  (${ov.rows.toLocaleString()}, ${ov.columns})`, 'value'],
                [`# Numeric cols    →  ${numCols.length}`, ''],
                ...numCols.slice(0, 4).map(c => [`  • ${c}  (float64/int64)`, '']),
                [`# Categorical     →  ${catCols.length}`, ''],
                ...catCols.slice(0, 4).map(c => [`  • ${c}  (object)`, '']),
                ...(dtCols.length ? [`# DateTime cols   →  ${dtCols.length}`] : []).map(t => [t, '']),
                ['✔ Structure mapped', 'success'],
            ]},

            // Phase 3 – Missing
            { delay: 2400, phase: '[ 3/7 ] Missing Value Analysis', pct: 35, lines: (() => {
                const lines = [['', ''], ['# ── df.isnull().sum() ───────────────', 'dim']];
                const missCols = Object.entries(mp).filter(([,v]) => v.missing_before > 0);
                if (missCols.length === 0) {
                    lines.push(['# All columns complete — 0 nulls', 'success']);
                } else {
                    Object.entries(mp).slice(0, 8).forEach(([col, v]) => {
                        const flag = v.missing_before > 0 ? '  ⚠' : '  ✔';
                        const cls  = v.missing_before > 0 ? 'warn' : 'success';
                        lines.push([`  ${col.padEnd(22)} →  ${v.missing_before}${flag}`, cls]);
                    });
                }
                lines.push(['', '']);
                lines.push(['# ── Imputation ──────────────────────', 'dim']);
                missCols.slice(0, 4).forEach(([col, v]) => {
                    lines.push([`df['${col}'].fillna(${v.fill_value || 'mode'})  # ${v.method}`, 'value']);
                });
                const totalFixed = Object.values(mp).reduce((a, v) => a + v.missing_before, 0);
                lines.push([`✔ ${totalFixed} missing values handled`, 'success']);
                return lines;
            })()},

            // Phase 4 – Stats
            { delay: 4200, phase: '[ 4/7 ] Descriptive Statistics', pct: 52, lines: (() => {
                const lines = [['', ''], ['# ── df.describe() ───────────────────', 'dim']];
                numCols.slice(0, 3).forEach(col => {
                    const s = stats[col];
                    if (!s) return;
                    lines.push([`# ${col}`, 'dim']);
                    lines.push([`  mean=${s.mean}  std=${s.std}  min=${s.min}  max=${s.max}`, '']);
                    lines.push([`  25%=${s['25%']}  75%=${s['75%']}  skew=${s.skewness}`, '']);
                    if (Math.abs(s.skewness) > 1) lines.push([`  ⚠ Skewed ${s.skewness > 0 ? 'right' : 'left'} (skew=${s.skewness})`, 'warn']);
                });
                lines.push([`✔ Stats for ${numCols.length} numeric columns`, 'success']);
                return lines;
            })()},

            // Phase 5 – Outliers
            { delay: 6200, phase: '[ 5/7 ] Outlier Detection (IQR)', pct: 67, lines: (() => {
                const lines = [['', ''], ['# ── IQR Method ──────────────────────', 'dim']];
                const oCols = Object.entries(outliers).filter(([,v]) => v.outliers_count > 0);
                if (oCols.length === 0) {
                    lines.push(['# No outliers detected ✔', 'success']);
                } else {
                    oCols.slice(0, 4).forEach(([col, v]) => {
                        lines.push([`# ${col}`, 'dim']);
                        lines.push([`  Q1=${v.Q1}  Q3=${v.Q3}  IQR=${v.IQR}`, '']);
                        lines.push([`  lower=${v.lower_bound}  upper=${v.upper_bound}`, '']);
                        lines.push([`  outliers: ${v.outliers_count} rows  ⚠`, 'warn']);
                    });
                }
                const totalO = Object.values(outliers).reduce((a,v) => a + v.outliers_count, 0);
                lines.push([`✔ ${totalO} total outlier rows flagged`, totalO > 0 ? 'warn' : 'success']);
                return lines;
            })()},

            // Phase 6 – Correlation
            { delay: 8000, phase: '[ 6/7 ] Correlation Matrix', pct: 82, lines: (() => {
                const lines = [['', ''], ['# ── Pearson Correlation ─────────────', 'dim'],
                    ['corr = df.select_dtypes(np.number).corr()', 'value']];
                const corrCols = Object.keys(corr).slice(0, 4);
                if (corrCols.length >= 2) {
                    // header
                    lines.push([('       ' + corrCols.map(c => c.slice(0,8).padEnd(9)).join('')), 'dim']);
                    corrCols.forEach(row => {
                        const cells = corrCols.map(col => {
                            const v = corr[row]?.[col] ?? 0;
                            return v.toFixed(2).padEnd(9);
                        }).join('');
                        const highlight = corrCols.some(col => col !== row && Math.abs(corr[row]?.[col] ?? 0) >= 0.7);
                        lines.push([`${row.slice(0,7).padEnd(7)}${cells}`, highlight ? 'success' : '']);
                    });
                } else {
                    lines.push(['# < 2 numeric columns — skipped', 'dim']);
                }
                lines.push(['✔ Correlation matrix computed', 'success']);
                return lines;
            })()},

            // Phase 7 – Visualizations + Insights
            { delay: 9800, phase: '[ 7/7 ] Visualizations & Insights', pct: 96, lines: (() => {
                const lines = [['', ''], ['# ── Chart Generation ────────────────', 'dim']];
                numCols.slice(0, 3).forEach(col => {
                    lines.push([`plt.hist(df['${col}'], bins=20)     ✔`, 'value']);
                });
                catCols.slice(0, 2).forEach(col => {
                    lines.push([`df['${col}'].value_counts().plot()   ✔`, 'value']);
                });
                if (Object.keys(corr).length >= 2) {
                    lines.push(['sns.heatmap(corr, annot=True)         ✔', 'value']);
                }
                lines.push(['', '']);
                lines.push(['# ── Auto Insights ───────────────────', 'dim']);
                d.insights.slice(0, 6).forEach(ins => lines.push([`  ${ins}`, 'success']));
                lines.push(['', '']);
                lines.push(['# ════════════════════════════════════', 'dim']);
                lines.push([`#  ✔ EDA COMPLETE  ${ov.rows.toLocaleString()} rows analyzed  `, 'success']);
                lines.push(['# ════════════════════════════════════', 'dim']);
                return lines;
            })()},
        ];

        streamPhases(phases, box, 10800);
    }

    function streamPhases(phases, box, totalDuration) {
        phases.forEach(phase => {
            setTimeout(() => {
                setPhase(phase.phase, phase.pct);
                phase.lines.forEach((line, i) => {
                    setTimeout(() => {
                        addCodeLine(line[0], line[1], box);
                    }, i * 75);
                });
            }, phase.delay);
        });
        setTimeout(() => {
            setPhase('✔ EDA COMPLETE — All stages done', 100);
        }, totalDuration);
    }

    function renderQuality(data) {
        const mp = data.missing_handling_process;
        const dupes = data.data_quality?.duplicates || 0;
        const missingCols = Object.values(mp).filter(v => v.missing_before > 0).length;

        document.getElementById('q-dupes').textContent = dupes;
        document.getElementById('q-missing-cols').textContent = missingCols;

        const tbody = document.getElementById('cleaningBody');
        tbody.innerHTML = '';

        Object.entries(mp).forEach(([col, val]) => {
            const fixed = val.missing_before > val.missing_after;
            const status = val.missing_before === 0 ? '✔️ Clean' : fixed ? '✔️ Fixed' : '⚠️ Remaining';

            tbody.innerHTML += `
                <tr>
                    <td style="color: var(--accent);">${col}</td>
                    <td>${val.col_type}</td>
                    <td style="color: var(--red);">${val.missing_before}</td>
                    <td style="color: ${val.missing_after > 0 ? 'var(--yellow)' : 'var(--green)'};">${val.missing_after}</td>
                    <td>${val.fill_value || '—'}</td>
                    <td>${val.method || '—'}</td>
                    <td style="color: ${val.missing_before === 0 || fixed ? 'var(--green)' : 'var(--yellow)'};">${status}</td>
                </tr>
            `;
        });

        renderMissingPanel('before');
    }

    function renderMissingPanel(view) {
        const panelDiv = document.getElementById('missingPanel');
        const btnBefore = document.getElementById('btnMissingBefore');
        const btnHandling = document.getElementById('btnMissingHandling');
        const btnAfter = document.getElementById('btnMissingAfter');

        [btnBefore, btnHandling, btnAfter].forEach(b => b.classList.remove('active'));

        if (view === 'before') {
            btnBefore.classList.add('active');
            renderMissingBefore(globalData, panelDiv);
        } else if (view === 'handling') {
            btnHandling.classList.add('active');
            renderMissingHandling(globalData, panelDiv);
        } else {
            btnAfter.classList.add('active');
            renderMissingAfter(globalData, panelDiv);
        }
    }

            <div class="mv-view-title">✅ DATASET STATE AFTER CLEANING</div>
            <div class="mv-summary">
                <div class="mv-chip green">✔ ${totalFixed} columns fully fixed</div>
                ${totalRemaining > 0 ? `<div class="mv-chip yellow">⚠ ${totalRemaining} columns still have missing</div>` : ''}
                <div class="mv-chip accent2">${rows.toLocaleString()} rows preserved</div>
            </div>
        `;

        if (missingEntries.length === 0) {
            html += `<div class="mv-no-missing">✅ No missing values — no cleaning needed!</div>`;
            panelDiv.innerHTML = html;
            return;
        }

        missingEntries.forEach(([col, val]) => {
            const isFixed   = val.missing_after === 0;
            const statusColor = isFixed ? 'var(--green)' : 'var(--yellow)';
            const statusIcon  = isFixed ? '✅' : '⚠️';
            const statusText  = isFixed ? 'FIXED' : 'REMAINING';

            const beforePct = val.missing_pct !== undefined ? val.missing_pct.toFixed(1) : ((val.missing_before / rows) * 100).toFixed(1);
            const afterPct  = val.missing_after > 0 ? ((val.missing_after / rows) * 100).toFixed(1) : '0.0';

            // Value count tables
            const vcBefore  = val.vc_before || {};
            const vcAfter   = val.vc_after  || {};
            const fillStrat = val.fill_strategy || val.method || '—';
            const fillVal   = val.fill_value;
            const typeBadge = val.col_type === 'Numeric' ? 'numeric' : 'cat';

            // Build vc_before table
            let vcBeforeHtml = '';
            Object.entries(vcBefore).filter(([k]) => k !== 'nan' && k !== 'NaN' && k !== 'None').slice(0, 5).forEach(([k, cnt]) => {
                vcBeforeHtml += `<div class="mv-vc-item"><span class="mv-vc-val">${k}</span><span class="mv-vc-cnt">${cnt}</span></div>`;
            });

            // Build vc_after table
            let vcAfterHtml = '';
            Object.entries(vcAfter).filter(([k]) => k !== 'nan' && k !== 'NaN' && k !== 'None').slice(0, 5).forEach(([k, cnt]) => {
                vcAfterHtml += `<div class="mv-vc-item"><span class="mv-vc-val">${k}</span><span class="mv-vc-cnt">${cnt}</span></div>`;
            });

            html += `
                <div class="mv-block">
                    <!-- Header row -->
                    <div class="mv-col-header">
                        <span class="mv-col-name">${col}</span>
                        <div style="display:flex;gap:6px;align-items:center;flex-wrap:wrap;">
                            <span class="mv-col-badge ${typeBadge}">${val.col_type}</span>
                            <span style="font-family:var(--font-mono);font-size:11px;color:${statusColor};">
                                ${statusIcon} ${statusText}
                            </span>
                        </div>
                    </div>

                    <!-- Before → Handling → After 3-column grid -->
                    <div class="mv-after-grid">

                        <!-- BEFORE column -->
                        <div class="mv-after-cell">
                            <div class="mv-after-cell-title" style="color:var(--red);">📋 Before</div>
                            <div style="color:var(--red);font-family:var(--font-mono);font-size:14px;margin-bottom:6px;">
                                ${val.missing_before.toLocaleString()}
                                <span style="font-size:10px;"> (${beforePct}%)</span>
                            </div>
                            <div class="mv-bar-track" style="margin-bottom:8px;">
                                <div class="mv-bar-fill red" style="width:${Math.min(100, parseFloat(beforePct))}%"></div>
                            </div>
                            ${vcBeforeHtml ? `<div style="margin-top:6px;">${vcBeforeHtml}</div>` : '<div style="color:var(--text-dim);font-size:10px;">No data</div>'}
                        </div>

                        <!-- HANDLING column -->
                        <div class="mv-after-cell">
                            <div class="mv-after-cell-title" style="color:var(--accent2);">⚙️ Handling</div>
                            <div style="color:var(--accent);font-size:12px;line-height:1.5;margin-bottom:6px;">
                                ${fillStrat}
                            </div>
                            ${fillVal !== null && fillVal !== undefined ? `
                            <div style="margin-top:4px;">
                                <span style="font-family:var(--font-mono);font-size:10px;color:var(--text-dim);">Fill value:</span>
                                <span style="font-family:var(--font-mono);font-size:12px;color:var(--yellow);"> ${fillVal}</span>
                            </div>` : ''}
                            ${val.indicator_col ? `
                            <div style="margin-top:8px;">
                                <span class="mv-indicator-tag">${val.indicator_col}</span>
                                <div style="font-size:10px;color:var(--text-dim);margin-top:4px;">added</div>
                            </div>` : ''}
                        </div>

                        <!-- AFTER column -->
                        <div class="mv-after-cell">
                            <div class="mv-after-cell-title" style="color:${statusColor};">✅ After</div>
                            <div style="color:${isFixed ? 'var(--green)' : 'var(--yellow)'};font-family:var(--font-mono);font-size:14px;margin-bottom:6px;">
                                ${val.missing_after.toLocaleString()}
                                <span style="font-size:10px;"> (${afterPct}%)</span>
                            </div>
                            <div class="mv-bar-track" style="margin-bottom:8px;">
                                <div class="mv-bar-fill ${isFixed ? 'green' : 'yellow'}" style="width:${Math.min(100, parseFloat(afterPct))}%"></div>
                            </div>
                            ${vcAfterHtml ? `<div style="margin-top:6px;">${vcAfterHtml}</div>` : '<div style="color:var(--text-dim);font-size:10px;">No data</div>'}
                        </div>

                    </div><!-- end mv-after-grid -->
                </div>
            `;
        });

        panelDiv.innerHTML = html;
    }

    function generatePivotTable() {
        if (!globalData) return;

        const rowCol = document.getElementById('pivotRows').value;
        const colCol = document.getElementById('pivotCols').value;
        const valCol = document.getElementById('pivotValue').value;

        const result = {};
        globalData.preview.forEach(row => {
            const rKey = row[rowCol];
            const cKey = row[colCol];
            const val = parseFloat(row[valCol]) || 0;

            if (!result[rKey]) result[rKey] = {};
            if (!result[rKey][cKey]) result[rKey][cKey] = 0;
            result[rKey][cKey] += val;
        });

        const cols = [...new Set(globalData.preview.map(r => r[colCol]))];
        let html = `<table><thead><tr><th>${rowCol}</th>`;
        cols.forEach(c => html += `<th>${c}</th>`);
        html += '</tr></thead><tbody>';

        Object.entries(result).forEach(([rKey, rVal]) => {
            html += `<tr><td style="color: var(--accent);">${rKey}</td>`;
            cols.forEach(c => html += `<td>${(rVal[c] || 0).toFixed(2)}</td>`);
            html += '</tr>';
        });

        html += '</tbody></table>';
        document.getElementById('pivotTableResult').innerHTML = html;
        showToast('✅ Pivot generated');
    }

    function renderValueCounts(data) {
        const sel = document.getElementById('vcColSelect');
        sel.innerHTML = Object.keys(data.value_counts).map(c => `<option value="${c}">${c}</option>`).join('');

        // Remove old listener to prevent duplicates on re-render
        const newSel = sel.cloneNode(true);
        sel.parentNode.replaceChild(newSel, sel);

        newSel.addEventListener('change', () => {
            const col = newSel.value;
            const counts = data.value_counts[col];
            if (!counts) return;
            const total = Object.values(counts).reduce((a, b) => a + b, 0);

            document.getElementById('vcTitle').textContent = `Distribution — ${col}`;

            const tbody = document.getElementById('vcBody');
            tbody.innerHTML = '';

            Object.entries(counts)
                .sort((a, b) => b[1] - a[1])
                .slice(0, 50)
                .forEach(([val, cnt]) => {
                    const pct = ((cnt / total) * 100).toFixed(1);
                    tbody.innerHTML += `<tr><td>${val}</td><td>${cnt}</td><td>${pct}%</td></tr>`;
                });
        });

        if (newSel.options.length > 0) newSel.dispatchEvent(new Event('change'));
    }

    let _vdMode  = 1;       // 1=head, 2=tail, 3=missing, 4=custom

    function _syncVdUI() {
        [1,2,3,4].forEach(i => {
            document.getElementById('vdCard' + i)?.classList.toggle('active', i === _vdMode);
        });

        const subBar    = document.getElementById('vdSubBar');
        const customBar = document.getElementById('vdCustomBar');
        const subLabel  = document.getElementById('vdSubLabel');

        if (_vdMode === 4) {
            subBar.style.display    = 'none';
            customBar.style.display = 'block';
            return;
        }
        subBar.style.display    = 'flex';
        customBar.style.display = 'none';

        const labels = {
            1: 'df.head() — view:',
            2: 'df.tail() — view:',
            3: 'Missing filter — rows with null/NaN:',
        };
        if (subLabel) subLabel.textContent = labels[_vdMode] || '';

        const pillColor = _vdMode === 2 ? 'accent2' : _vdMode === 3 ? 'warn' : '';
        ['100','75','50','25'].forEach((s, i) => {
            const pill = document.getElementById('vdPill' + (i + 1));
            if (!pill) return;
            if (_vdMode === 3) {
                pill.style.display = 'none';
            } else {
                pill.style.display = '';
                pill.className = 'vd-pill' + (pillColor ? ' ' + pillColor : '') + (_vdSlice === s ? ' active' : '');
            }
        });

        if (_vdMode === 3 && subLabel) {
            subLabel.textContent = 'Showing only rows with missing / null values';
        }
    }

    function vdRenderTable() {
        const wrap = document.getElementById('vdTableWrap');
        if (!_vdData) {
            wrap.innerHTML = '<div class="vd-empty-state"><div class="vd-empty-icon">📂</div><div>Upload a dataset to view it here</div></div>';
            return;
        }

        const ds     = _vdData.dataset_slices;
        const mp     = _vdData.missing_handling_process || {};
        const totalN = _vdData.overview.rows;

        // ── Select rows ──
        let rows = [];
        let modeLabel = '';
        let methodLabel = '';

        if (_vdMode === 1) {
            rows        = (ds?.head?.[_vdSlice]) || [];
            modeLabel   = `df.head(${_vdSlice}%) — First ${rows.length.toLocaleString()} of ${totalN.toLocaleString()} rows`;
            methodLabel = `pd.set_option('display.max_rows', None) · df.head(${rows.length})`;
        } else if (_vdMode === 2) {
            rows        = (ds?.tail?.[_vdSlice]) || [];
            modeLabel   = `df.tail(${_vdSlice}%) — Last ${rows.length.toLocaleString()} of ${totalN.toLocaleString()} rows`;
            methodLabel = `pd.set_option('display.max_rows', None) · df.tail(${rows.length})`;
        } else if (_vdMode === 3) {
            rows        = ds?.missing || [];
            modeLabel   = `Missing rows — ${rows.length.toLocaleString()} rows contain null/NaN`;
            methodLabel = `df[df.isnull().any(axis=1)]`;
        } else if (_vdMode === 4) {
            const start   = parseInt(document.getElementById('vdRangeStart')?.value) || 0;
            const end     = parseInt(document.getElementById('vdRangeEnd')?.value)   || 100;
            const allRows = ds?.head?.['100'] || [];
            rows          = allRows.slice(start, end);
            modeLabel     = `Custom slice — rows ${start} to ${Math.min(end, allRows.length)} of ${totalN.toLocaleString()}`;
            methodLabel   = `with pd.option_context('display.max_rows', None, 'display.max_columns', None, 'display.max_colwidth', None): display(df.iloc[${start}:${end}])`;
        }

        const allCols       = rows.length > 0 ? Object.keys(rows[0]) : [];
        const missingColSet = new Set(Object.keys(mp).filter(c => mp[c].missing_before > 0));

        // ── Stats bar ──
        const statsBar = document.getElementById('vdStatsBar');
        if (statsBar) {
            const missCells = _vdMode === 3
                ? rows.reduce((a, row) => a + allCols.filter(c => {
                    const v = row[c];
                    return v === null || v === undefined || String(v).trim().toLowerCase() === 'nan' || v === '';
                  }).length, 0)
                : 0;

            statsBar.innerHTML = `
                <div class="vd-stat-chip">Rows shown <span>${rows.length.toLocaleString()}</span></div>
                <div class="vd-stat-chip">Total rows <span>${totalN.toLocaleString()}</span></div>
                <div class="vd-stat-chip">Columns <span>${allCols.length}</span></div>
                <div class="vd-stat-chip" style="font-family:var(--font-mono);font-size:10px;color:var(--accent2);flex:1;">
                    <span style="color:var(--text-dim)">${methodLabel}</span>
                </div>
                ${_vdMode === 3 ? `<div class="vd-stat-chip red">Missing cells <span>${missCells.toLocaleString()}</span></div>` : ''}
                ${missingColSet.size > 0
                    ? `<div class="vd-stat-chip warn">Cols w/ missing <span>${missingColSet.size}</span></div>`
                    : '<div class="vd-stat-chip success">No missing cols <span>✔</span></div>'}
            `;
        }

        // ── Empty state ──
        if (rows.length === 0) {
            wrap.innerHTML = `
                <div class="vd-empty-state">
                    <div class="vd-empty-icon">${_vdMode === 3 ? '✅' : '📭'}</div>
                    <div>${_vdMode === 3
                        ? 'No missing values found — your dataset is completely clean!'
                        : 'No rows to display for this selection.'}</div>
                </div>`;
            return;
        }

        // ── Build table ──
        let html = `<table class="vd-table">`;

        // Header
        html += '<thead><tr><th class="row-idx">#</th>';
        allCols.forEach(col => {
            const hasMissing = missingColSet.has(col);
            html += `<th style="${hasMissing ? 'color:var(--yellow)' : ''}">${col}${hasMissing ? '<span class="vd-missing-badge">⚠</span>' : ''}</th>`;
        });
        html += '</tr></thead>';

        // Body
        html += '<tbody>';
        rows.forEach((row, ri) => {
            html += '<tr>';
            html += `<td class="row-idx">${ri + 1}</td>`;
            allCols.forEach(col => {
                const val     = row[col];
                const isEmpty = val === null || val === undefined || val === ''
                             || String(val).trim().toLowerCase() === 'nan'
                             || String(val).trim().toLowerCase() === 'none';
                const rawStr  = isEmpty ? '' : String(val);
                const display = isEmpty ? '<span style="color:var(--red);opacity:0.7">⬜ null</span>'
                                        : rawStr.length > 35 ? rawStr.slice(0, 35) + '…' : rawStr;
                const cls     = isEmpty ? 'missing-cell' : '';
                html += `<td class="${cls}" title="${isEmpty ? 'Missing / null value' : rawStr}">${display}</td>`;
            });
            html += '</tr>';
        });
        html += '</tbody></table>';

        // Code snippet bar at bottom
        html += `<div style="padding:10px 14px;background:rgba(0,0,0,0.2);border-top:1px solid var(--border);font-family:var(--font-mono);font-size:10px;color:var(--text-dim);">
            <span style="color:var(--accent2)">▶</span>&nbsp;
            <span style="color:var(--accent)">with</span> pd.option_context(
            <span style="color:var(--yellow)">'display.max_rows'</span>, None,
            <span style="color:var(--yellow)">'display.max_columns'</span>, None,
            <span style="color:var(--yellow)">'display.max_colwidth'</span>, None):
            &nbsp;<span style="color:var(--green)">display</span>(df)
            &nbsp;&nbsp;<span style="color:var(--text-dim)">→ ${rows.length.toLocaleString()} rows displayed</span>
        </div>`;

        wrap.innerHTML = html;
    }

    function openAuthModal() {
        document.getElementById('authModal').classList.add('active');
        switchAuthTab('login');
    }

    function switchAuthTab(tab) {
        document.getElementById('loginForm').style.display    = tab === 'login'    ? 'block' : 'none';
        document.getElementById('registerForm').style.display = tab === 'register' ? 'block' : 'none';
        document.getElementById('tabLogin').classList.toggle('active',    tab === 'login');
        document.getElementById('tabRegister').classList.toggle('active', tab === 'register');
        // clear messages
        ['loginError','loginSuccess','regError','regSuccess'].forEach(id => {
            const el = document.getElementById(id);
            if (el) { el.style.display = 'none'; el.textContent = ''; }
        });
    }

    async function authLogin() {
        const email    = document.getElementById('loginEmail').value.trim();
        const password = document.getElementById('loginPassword').value;
        const errEl    = document.getElementById('loginError');
        const okEl     = document.getElementById('loginSuccess');

        if (!email || !password) {
            errEl.textContent = '⚠ Fill in all fields.'; errEl.style.display = 'block'; return;
        }

        try {
            const res  = await fetch('/login', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ email, password })
            });
            const data = await res.json();

            if (!res.ok) {
                errEl.textContent = '❌ ' + (data.error || 'Login failed.'); errEl.style.display = 'block'; return;
            }

            authToken = data.token;
            authUser  = { id: data.user_id, name: data.name, email };
            okEl.textContent = '✅ Logged in!'; okEl.style.display = 'block';
            setTimeout(() => { closeModal('authModal'); updateAuthUI(); showToast('✅ Welcome, ' + data.name); }, 800);

        } catch(e) {
            errEl.textContent = '❌ Server error.'; errEl.style.display = 'block';
        }
    }

    async function authRegister() {
        const name     = document.getElementById('regName').value.trim();
        const email    = document.getElementById('regEmail').value.trim();
        const password = document.getElementById('regPassword').value;
        const errEl    = document.getElementById('regError');
        const okEl     = document.getElementById('regSuccess');

        if (!name || !email || !password) {
            errEl.textContent = '⚠ Fill in all fields.'; errEl.style.display = 'block'; return;
        }

        try {
            const res  = await fetch('/register', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ name, email, password })
            });
            const data = await res.json();

            if (!res.ok) {
                errEl.textContent = '❌ ' + (data.error || 'Registration failed.'); errEl.style.display = 'block'; return;
            }

            authToken = data.token;
            authUser  = { id: data.user_id, name, email };
            okEl.textContent = '✅ Account created!'; okEl.style.display = 'block';
            setTimeout(() => { closeModal('authModal'); updateAuthUI(); showToast('✅ Welcome, ' + name); }, 800);

        } catch(e) {
            errEl.textContent = '❌ Server error.'; errEl.style.display = 'block';
        }
    }

    function authLogout() {
        authToken = null;
        authUser  = null;
        updateAuthUI();
        showToast('👋 Logged out');
    }

    function updateAuthUI() {
        const badge    = document.getElementById('userBadge');
        const loginBtn = document.getElementById('loginBtnSidebar');

        if (authUser) {
            badge.style.display = 'block';
            loginBtn.style.display = 'none';
            document.getElementById('sidebarUserName').textContent  = '👤 ' + authUser.name;
            document.getElementById('sidebarUserEmail').textContent = authUser.email;
        } else {
            badge.style.display = 'none';
            loginBtn.style.display = 'block';
        }
    }

    async function loadFileHistory() {
        const noticeEl = document.getElementById('historyLoginNotice');
        const tableCard = document.getElementById('historyTableCard');
        const tbody = document.getElementById('historyTableBody');

        if (!authToken) {
            noticeEl.style.display = 'block';
            tableCard.style.display = 'none';
            return;
        }

        noticeEl.style.display = 'none';
        tableCard.style.display = 'block';
        tbody.innerHTML = '<tr><td colspan="5" style="color:var(--text-dim);text-align:center;">Loading...</td></tr>';

        try {
            const res  = await fetch('/files/history', {
                headers: { 'Authorization': 'Bearer ' + authToken }
            });
            const data = await res.json();

            if (!res.ok) {
                tbody.innerHTML = '<tr><td colspan="5" style="color:var(--red);">Failed to load history.</td></tr>';
                return;
            }

            const files = data.files || [];
            if (files.length === 0) {
                tbody.innerHTML = '<tr><td colspan="5" style="color:var(--text-dim);text-align:center;">No files uploaded yet.</td></tr>';
                return;
            }

            tbody.innerHTML = files.map((f, i) => `
                <tr>
                    <td>${i + 1}</td>
                    <td style="color:var(--accent);">${f.file_name}</td>
                    <td>${f.file_type.toUpperCase()}</td>
                    <td>${(f.file_size / 1024).toFixed(1)} KB</td>
                    <td>${new Date(f.uploaded_at).toLocaleString()}</td>
                </tr>
            `).join('');

        } catch(e) {
            tbody.innerHTML = '<tr><td colspan="5" style="color:var(--red);">Error loading history.</td></tr>';
        }
    }
